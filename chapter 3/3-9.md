- 请重新编写以下程序，要求使用嵌套if-else语句，并测试执行；

```c
int main(void){
    double x,y;
    printf("Enter x:");
    scanf("%lf", &x);
    if(x<0){
        y = 0;
    }else if(x<=15){
        y = 4 * x / 3;
    }else{
        y = 2.5 * x - 10.5;
    }
    printf("f(%.2f) = %.2f\n", x, y);
    return 0;
}
```

- 使用嵌套if-else语句，重新编写
```c
int main(void){
    double x, y;
    printf("Enter x:");
    scanf("%lf", &x);

    if(x < 0){
        y = 0;
    } else {
        if(x <= 15){
            y = 4 * x / 3;
        } else {
            y = 2.5 * x - 10.5;
        }
    }
    
    printf("f(%.2f) = %.2f\n", x, y);
    return 0;
}
```
- 执行结果
```

f(-10.00) = 0.00

Enter x:0

Enter x:25

```
